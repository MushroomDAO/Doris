name: Deploy Blog

on:
  push:
    branches: [ main ]
    paths: 
      - 'docs/**'
      - 'templates/**'
      - 'scripts/**'
  workflow_dispatch:

permissions:
  contents: read
  pages: write
  id-token: write

concurrency:
  group: "pages"
  cancel-in-progress: false

jobs:
  # AI å†…å®¹å¢žå¼º
  ai-enhance:
    runs-on: ubuntu-latest
    if: contains(github.event.head_commit.message, '[ai-enhance]')
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          token: ${{ secrets.GITHUB_TOKEN }}

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: AI Content Enhancement
        env:
          OPENAI_API_KEY: ${{ secrets.OPENAI_API_KEY }}
          ANTHROPIC_API_KEY: ${{ secrets.ANTHROPIC_API_KEY }}
        run: |
          node scripts/ai-enhance.js
          
      - name: Commit enhanced content
        run: |
          git config --local user.email "action@github.com"
          git config --local user.name "GitHub Action"
          git add docs/
          if [ -n "$(git diff --staged)" ]; then
            git commit -m "ðŸ¤– AI enhanced content [skip ci]"
            git push
          fi

  # æž„å»ºå’Œéƒ¨ç½²
  build-and-deploy:
    runs-on: ubuntu-latest
    needs: [ai-enhance]
    if: always() && !failure()
    environment:
      name: github-pages
      url: ${{ steps.deployment.outputs.page_url }}
    
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Build
        run: |
          npm run build
          
      - name: Setup Pages
        uses: actions/configure-pages@v4
        
      - name: Upload to Pages
        uses: actions/upload-pages-artifact@v3
        with:
          path: './docs'
          
      - name: Deploy to GitHub Pages
        id: deployment
        uses: actions/deploy-pages@v4

  # IPFS éƒ¨ç½²
  deploy-ipfs:
    runs-on: ubuntu-latest
    needs: [build-and-deploy]
    if: success()
    
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '18'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Deploy to IPFS
        env:
          PINATA_API_KEY: ${{ secrets.PINATA_API_KEY }}
          PINATA_SECRET_API_KEY: ${{ secrets.PINATA_SECRET_API_KEY }}
          WEB3_STORAGE_TOKEN: ${{ secrets.WEB3_STORAGE_TOKEN }}
        run: |
          node scripts/deploy-ipfs.js
          
      - name: Update IPFS hash
        run: |
          git config --local user.email "action@github.com"
          git config --local user.name "GitHub Action"
          if [ -f "ipfs-hash.txt" ]; then
            git add ipfs-hash.txt
            git commit -m "ðŸ“¡ Update IPFS hash [skip ci]" || exit 0
            git push
          fi
